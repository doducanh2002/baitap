import java.util.Scanner;
public class dfafa {
public static int TimSoLon(int [] mang) {
    int demMax=0;
    for (int i = 0; i < mang.length; i++) {
        int soLonNhat=mang[0];
        if (mang[i] > soLonNhat) {
            soLonNhat = mang[i];
            if (mang[i] == soLonNhat) {
                demMax = i + 1;
            }
        }
    }
    return demMax;
}
    public static int TimSoNho(int [] mang) {
        int demMin=0;
        for (int i = 0; i < mang.length; i++) {
            int soNhoNhat=mang.length;
            if (mang[i] < soNhoNhat) {
                soNhoNhat = mang[i];
                if (mang[i] == soNhoNhat) {
                    demMin = i + 1;
                }
            }
        }
        return demMin;
    }
